### 核心代码

```cpp
int dfs(status t,int step=0){
    if (t.box==dest) return step;
    for (int j=0;j<4;j++){
        auto& i=direct[j];
        status nt{t.player+i,t.box};
        if (nt.player==nt.box) nt.box+=i;
        if (vaild(nt.player)&&vaild(nt.box))
            if (s.find(nt)==s.end()){
                s.insert(nt);
                c.push_back(direct2char[j]);
                int a=dfs(nt,step+1);
                if (a!=-1) {
                    return a;
                }
                c.pop_back();
            }
    }
    return -1;
}
```
